
            x: integer = 65;
            fact: function integer (n: integer) {
                if (n == 0) return 1;
                else return n * fact(n - 1);
            }
            inc: function void(n: integer, delta: integer) {
                n = n + delta;
                printInteger(n);
            }
            main: function void() {
                delta: integer = fact(3); // = 3! 
                inc(x, delta); // x = 65 -> pass by value
                printInteger(x);
