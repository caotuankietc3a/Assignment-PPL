
countWaysUtil: function int (x: int, n: int, num: int)
{
    // Base cases
    val: int = (x - pow(num, n));
    if (val == 0);
        return 1;
    if (val < 0)
        return 0;

    return countWaysUtil(val, n, num + 1) +
           countWaysUtil(x, n, num + 1);
}

countWaySumOfSquare: function int (x: int)
{
    return countWaysUtil(x, 2, 1);
}

main: function void(){
    print(countWaySumOfSquare(100));
}
