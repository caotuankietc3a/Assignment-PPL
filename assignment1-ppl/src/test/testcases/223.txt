Fibonacci: function int(n: int) {
    f0,   f1,   fn: auto = 0, 1, 1;
    if (n < 0) {
        return -1;
    }
    if ((n == 0) || (n == 1)) {
        return n;
    } else {
        for (i = 2, i < n, i + 1) {
          f0 = f1;
          f1 = fn;
          fn = f0 + f1;
        }
    }
    return fn;
}
check_prime: function boolean (n : int) {
  if (n < 2)
    return false;

  for (i = 2, i <= sqrt(n), i+1) {
    if (n % i == 0)
      return false;
  }
  return true;
}
